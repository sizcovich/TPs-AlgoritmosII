itlista(categoria) it <- ultimo(todos)                               O(1) //iterador a raiz
id <- 1                                                               O(1)
agregarListaVacia(hijos)                                              O(1) // crea una lista vacia al final
itlistalista(lista(lista(categoria))) itlista <- ultimo(hijos)        O(1) // es un it a la ultima lista agregada 
crearTrie(r, <id, it, itlista>)                                       O(|r|) //it es iterador del ultimo e itLista es el de hijos
raiz <- r                                                             O(|r|) //a la variable raiz le asigno la categoria

iagregarHijo(in h:categoria, in p:categoria, in/out: a:abcat)
agregarAtras(todos, h)                                                O(|h|)
itlista(categoria) it <- ultimo(todos)                                O(1)
id++                                                                  O(1)
agregarAtras(hijos)                                                   O(1)
itlistalista(lista(lista(categoria))) itlista <- ultimo(hijos)        O(1)
agregarTrie(a.trie, h, <id ,it, itlista>)                             O(|h|) //ver si la tupla es lo mas facil de pasar y orden
itlistalista(lista(lista(categoria))) itlistahijos                    O(1) //crea iterador vacio
itlistahijos <- buscarListaHijos(a.trie, p)                           O(|p|) //recorre el trie buscando al padre, llega al nodo y devuelve el iterador a la lista de hijos (buscarListaHijos(in trie, in categoria) -> it listalistaCategoria)
agregarAtras(actual(itlistahijos), h)                                 O(|h|) //a la lista de hijos le agregas el hijo que queres agregar
